REPORT ZVV_BED_ALLOT.
TABLES: zvv_bed,
        zvv_patient.

TYPE-POOLS: slis.

SELECT-OPTIONS:
  s_bedid FOR zvv_bed-bed_id,
  s_patid FOR zvv_bed-allotted_to,
  s_admit FOR zvv_bed-admitted_on.
PARAMETERS:
  p_ward TYPE zvv_bed-ward_name,
  p_dept TYPE zvv_bed-department_code.

DATA: lt_list   TYPE slis_t_listheader,
      ls_list   TYPE slis_listheader.
TYPES: BEGIN OF ty_final,

         bed_id          TYPE zvv_bed-bed_id,
         ward_name       TYPE zvv_bed-ward_name,
         department_code TYPE zvv_bed-department_code,
         allotted_to     TYPE zvv_bed-allotted_to,
         admitted_on     TYPE zvv_bed-admitted_on,
         discharge_date  TYPE zvv_bed-discharge_date,

         first_name      TYPE zvv_patient-first_name,
         last_name       TYPE zvv_patient-last_name,
         phone           TYPE zvv_patient-phone,
         email_id        TYPE zvv_patient-email_id,
         city            TYPE zvv_patient-city,
         region_code     TYPE zvv_patient-region_code,

       END OF ty_final.

DATA: gt_final TYPE TABLE OF ty_final,
      gs_final TYPE ty_final.

DATA: gt_fieldcat TYPE slis_t_fieldcat_alv,
      gs_fieldcat TYPE slis_fieldcat_alv.

DATA: gt_events TYPE slis_t_event,
      gs_event TYPE slis_alv_event.

START-OF-SELECTION.

  SELECT
         a~bed_id
         a~ward_name
         a~department_code
         a~allotted_to
         a~admitted_on
         a~discharge_date

         b~first_name
         b~last_name
         b~phone
         b~email_id
         b~city
         b~region_code

    INTO TABLE gt_final

    FROM zvv_bed AS a
    INNER JOIN zvv_patient AS b
      ON a~allotted_to = b~patient_id

    WHERE a~bed_id IN s_bedid
      AND a~allotted_to IN s_patid
      AND a~admitted_on IN s_admit.

  IF p_ward IS NOT INITIAL.
    DELETE gt_final WHERE ward_name <> p_ward.
  ENDIF.

  IF p_dept IS NOT INITIAL.
    DELETE gt_final WHERE department_code <> p_dept.
  ENDIF.

  IF gt_final IS INITIAL.
    MESSAGE 'No Data Found' TYPE 'I'.
    EXIT.
  ENDIF.



  ls_list-typ  = 'H'.
  ls_list-info = 'Patient Report'.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run Date'.
  ls_list-info = sy-datum.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run Time'.
  ls_list-info = sy-uzeit.
  APPEND ls_list TO lt_list.

  CLEAR ls_list.
  ls_list-typ  = 'S'.
  ls_list-key  = 'Run By'.
  ls_list-info = sy-uname.
  APPEND ls_list TO lt_list.

PERFORM build_fieldcat.
  CALL FUNCTION 'REUSE_ALV_GRID_DISPLAY'
    EXPORTING
       i_callback_program     = sy-repid
      it_fieldcat = gt_fieldcat
      it_events   = gt_events
       i_callback_top_of_page = 'TOP_OF_PAGE'
    TABLES
      t_outtab    = gt_final.
